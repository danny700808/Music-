    </section>

    <button class="nav-btn left" id="prevBtn" type="button" aria-label="上一頁">‹</button>
    <button class="nav-btn right" id="nextBtn" type="button" aria-label="下一頁">›</button>
  </main>

  <script>
    // 這裡已經改成你 GitHub 目前實際上傳的檔名，不需要重新上傳圖片。
    const pages = [
      "cover.jpg",
      "page01.jpg",
      "page02.jpg",
      "page03.jpg",
      "page04.jpg",
      "page05.jpg",
      "page06.jpg",
      "page07.jpg",
      "page08.jpg"
    ];

    const track = document.getElementById("track");
    const viewer = document.getElementById("viewer");
    const pageLabel = document.getElementById("pageLabel");
    const prevBtn = document.getElementById("prevBtn");
    const nextBtn = document.getElementById("nextBtn");

    let current = 0;
    let startX = 0;
    let currentX = 0;
    let isDragging = false;

    function buildSlides() {
      track.innerHTML = "";

      pages.forEach((src, index) => {
        const slide = document.createElement("div");
        slide.className = "slide" + (index === 0 ? " slide-cover" : "");

        const img = document.createElement("img");
        img.src = src;
        img.alt = index === 0 ? "節目單封面" : `節目單第 ${index} 頁`;

        img.onerror = () => {
          slide.innerHTML = `
            <div class="fallback">
              圖片讀取失敗<br>
              請確認檔名：${src}
            </div>
          `;
        };

        slide.appendChild(img);

        if (index === 0) {
          const cover = document.createElement("div");
          cover.className = "cover-content";
          cover.innerHTML = `
            <div class="cover-line"></div>
            <h1 class="cover-title">節目單</h1>
            <div class="cover-line bottom"></div>
            <div class="cover-subtitle">左右滑動查看節目</div>
            <div class="swipe-hint">
              <span>←</span>
              <span class="finger">☝️</span>
              <span>→</span>
            </div>
            <div class="dots" id="dots"></div>
          `;
          slide.appendChild(cover);
        }

        track.appendChild(slide);
      });

      buildDots();
      update();
    }

    function buildDots() {
      const dots = document.getElementById("dots");
      if (!dots) return;

      dots.innerHTML = "";
      pages.forEach((_, index) => {
        const dot = document.createElement("span");
        dot.className = "dot" + (index === current ? " active" : "");
        dots.appendChild(dot);
      });
    }

    function updateDots() {
      const dots = document.querySelectorAll(".dot");
      dots.forEach((dot, index) => {
        dot.classList.toggle("active", index === current);
      });
    }

    function update() {
      track.style.transform = `translateX(${-current * 100}%)`;
      pageLabel.textContent = current === 0 ? "封面" : `第 ${current} 頁`;
      prevBtn.disabled = current === 0;
      nextBtn.disabled = current === pages.length - 1;
      updateDots();
    }

    function goTo(index) {
      current = Math.max(0, Math.min(index, pages.length - 1));
      update();
    }

    prevBtn.addEventListener("click", () => goTo(current - 1));
    nextBtn.addEventListener("click", () => goTo(current + 1));

    viewer.addEventListener("touchstart", (e) => {
      startX = e.touches[0].clientX;
      currentX = startX;
      isDragging = true;
    }, { passive: true });

    viewer.addEventListener("touchmove", (e) => {
      if (!isDragging) return;
      currentX = e.touches[0].clientX;
    }, { passive: true });

    viewer.addEventListener("touchend", () => {
      if (!isDragging) return;
      const diff = currentX - startX;

      if (Math.abs(diff) > 55) {
        if (diff < 0) {
          goTo(current + 1);
        } else {
          goTo(current - 1);
        }
      }

      isDragging = false;
    });

    document.addEventListener("keydown", (e) => {
      if (e.key === "ArrowLeft") goTo(current - 1);
      if (e.key === "ArrowRight") goTo(current + 1);
    });

    buildSlides();
  </script>
